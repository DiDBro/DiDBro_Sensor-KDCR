\section{Kinematic Modeling}
% ===================================================================

\subsection{Joint Twists (Screw Axes)}

Using the \textbf{product of exponentials (PoE)} formulation, each joint is represented by a twist (screw axis) $\bm{\xi}_i \in \mathbb{R}^6$ expressed in the spatial frame $\{S\}$.

For a \textbf{revolute} joint with axis $\bm{\omega}$ and a point $\mathbf{q}$ on the axis:
\[
\bm{\xi}_i = \begin{bmatrix} -\hat{\bm{\omega}}_i\,\mathbf{q}_i \\[2pt] \bm{\omega}_i \end{bmatrix}
\qquad\text{where}\qquad
\text{hat}(\mathbf{v}) = \begin{bmatrix}
0 & -v_3 & v_2 \\
v_3 & 0 & -v_1 \\
-v_2 & v_1 & 0
\end{bmatrix}.
\]

For a \textbf{prismatic} joint with translation direction $\mathbf{v}$:
\[
\bm{\xi}_i = \begin{bmatrix} \mathbf{v}_i \\[2pt] \mathbf{0}_{3\times 1} \end{bmatrix}.
\]

\begin{conceptbox}[Joint Twists $\bm{\xi}_i$]
\begin{align*}
\text{Joint 1 (R, about $z$ at origin):}\quad
  \bm{\omega}_1 &= \begin{bmatrix}0\\0\\1\end{bmatrix},\;
  \mathbf{q}_1 = \begin{bmatrix}0\\0\\0\end{bmatrix}
  \;\Rightarrow\; \bm{\xi}_1 = \begin{bmatrix}0\\0\\0\\0\\0\\1\end{bmatrix} \\[5pt]
\text{Joint 2 (R, about $z$ at $[0,l_1,0]^\top$):}\quad
  \bm{\omega}_2 &= \begin{bmatrix}0\\0\\1\end{bmatrix},\;
  \mathbf{q}_2 = \begin{bmatrix}0\\l_1\\0\end{bmatrix}
  \;\Rightarrow\; \bm{\xi}_2 = \begin{bmatrix}l_1\\0\\0\\0\\0\\1\end{bmatrix} \\[5pt]
\text{Joint 3 (R, about $z$ at $[0,l_1+l_2,0]^\top$):}\quad
  \bm{\omega}_3 &= \begin{bmatrix}0\\0\\1\end{bmatrix},\;
  \mathbf{q}_3 = \begin{bmatrix}0\\l_1+l_2\\0\end{bmatrix}
  \;\Rightarrow\; \bm{\xi}_3 = \begin{bmatrix}l_1+l_2\\0\\0\\0\\0\\1\end{bmatrix} \\[5pt]
\text{Joint 4 (P, along $z$):}\quad
  \mathbf{v}_4 &= \begin{bmatrix}0\\0\\1\end{bmatrix}
  \;\Rightarrow\; \bm{\xi}_4 = \begin{bmatrix}0\\0\\1\\0\\0\\0\end{bmatrix}
\end{align*}
With numerical values $l_1 = 0.8$\,m, $l_2 = 0.6$\,m:
\[
\bm{\xi}_1 = \begin{bmatrix}0\\0\\0\\0\\0\\1\end{bmatrix},\quad
\bm{\xi}_2 = \begin{bmatrix}0.8\\0\\0\\0\\0\\1\end{bmatrix},\quad
\bm{\xi}_3 = \begin{bmatrix}1.4\\0\\0\\0\\0\\1\end{bmatrix},\quad
\bm{\xi}_4 = \begin{bmatrix}0\\0\\1\\0\\0\\0\end{bmatrix}.
\]
\end{conceptbox}

\subsection{Forward Kinematics via Product of Exponentials}

The homogeneous transformation from the spatial frame to link $i$ is:
\[
\mathbf{g}_{sl_i}(\bm{\theta}) = e^{\hat{\bm{\xi}}_1\theta_1}\,e^{\hat{\bm{\xi}}_2\theta_2}\cdots e^{\hat{\bm{\xi}}_i\theta_i}\;\mathbf{g}_{sl_i}(0)
\]
where $\mathbf{g}_{sl_i}(0)$ is the reference configuration of link $i$, and $\hat{\bm{\xi}}$ is the $4\times4$ matrix representation of twist $\bm{\xi}$ in $\mathfrak{se}(3)$:
\[
\hat{\bm{\xi}} = \begin{bmatrix} \hat{\bm{\omega}} & \mathbf{v} \\ \mathbf{0} & 0 \end{bmatrix}.
\]

\begin{stepbox}[Reference Configurations $\mathbf{g}_{sl_i}(0)$]
The reference (zero-angle) link frames are given by pure translations (no rotation):
\begin{align*}
\mathbf{g}_{sl_1}(0) &= \begin{bmatrix} \mathbf{I} & [0,\; r_1,\; 0.3]^\top \\ \mathbf{0} & 1 \end{bmatrix}, \quad r_1 = l_1/2 = 0.4 \\[3pt]
\mathbf{g}_{sl_2}(0) &= \begin{bmatrix} \mathbf{I} & [0,\; l_1+r_2,\; 0.4]^\top \\ \mathbf{0} & 1 \end{bmatrix}, \quad r_2 = l_2/2 = 0.3 \\[3pt]
\mathbf{g}_{sl_3}(0) &= \begin{bmatrix} \mathbf{I} & [0,\; l_1+l_2,\; 0.5]^\top \\ \mathbf{0} & 1 \end{bmatrix} \\[3pt]
\mathbf{g}_{sl_4}(0) &= \begin{bmatrix} \mathbf{I} & [0,\; l_1+l_2,\; l_0]^\top \\ \mathbf{0} & 1 \end{bmatrix}, \quad l_0 = 0.2
\end{align*}
\end{stepbox}

\subsection{\texorpdfstring{The $6\times 6$ Adjoint Transformation}{The 6x6 Adjoint Transformation}}

The adjoint map $\mathrm{Ad}_{\mathbf{g}} \in \mathbb{R}^{6\times 6}$ transforms twists between frames. For $\mathbf{g}=[\mathbf{R},\,\mathbf{p};\;\mathbf{0},\,1]$:
\[
\mathrm{Ad}_{\mathbf{g}} = \begin{bmatrix}
\mathbf{R} & \hat{\mathbf{p}}\mathbf{R} \\
\mathbf{0} & \mathbf{R}
\end{bmatrix}
\]
This is implemented in the code as \texttt{adjmg.m} and \texttt{MCNsim.m} (lines 160--172).

The \textbf{generalized inertia matrix} of link $i$ in its own body frame is the $6\times 6$ matrix:
\[
\hat{\mathbf{M}}_i = \begin{bmatrix}
m_i\mathbf{I}_3 & \mathbf{0} \\
\mathbf{0} & \mathbf{I}_i
\end{bmatrix},\qquad \mathbf{I}_i = \mathrm{diag}(I_{xi},\,I_{yi},\,I_{zi}).
\]

\subsection{Spatial Inertia Matrices}

Each body inertia is transformed to the spatial frame $\{S\}$ using the adjoint transformation:
\[
\hat{\mathbf{M}}_i^s = \mathrm{Ad}_{\mathbf{g}_{sl_i}(0)}^{-\top}\;\hat{\mathbf{M}}_i\;
\mathrm{Ad}_{\mathbf{g}_{sl_i}(0)}^{-1}.
\]
This is stored in the 3D array \texttt{calMp(:,:,i)}.

\subsection{Equations of Motion}

The manipulator dynamics in standard form:
\[
\boxed{\mathbf{M}(\bm{\theta})\,\ddot{\bm{\theta}} + \mathbf{C}(\bm{\theta},\dot{\bm{\theta}})\,\dot{\bm{\theta}} + \mathbf{N}(\bm{\theta},\dot{\bm{\theta}}) = \bm{\tau}}
\]
where:
\begin{itemize}
  \item $\mathbf{M}(\bm{\theta}) \in \mathbb{R}^{4\times 4}$ --- \textbf{mass/inertia matrix} (symmetric, positive definite)
  \item $\mathbf{C}(\bm{\theta},\dot{\bm{\theta}}) \in \mathbb{R}^{4\times 4}$ --- \textbf{Coriolis/centrifugal matrix}
  \item $\mathbf{N}(\bm{\theta},\dot{\bm{\theta}}) \in \mathbb{R}^{4}$ --- \textbf{gravity + friction vector}
  \item $\bm{\tau} \in \mathbb{R}^{4}$ --- \textbf{joint torques/forces}
\end{itemize}

\subsection{\texorpdfstring{Computation of $\mathbf{M}(\bm{\theta})$ via Screw Theory}{Computation of M(theta) via Screw Theory}}

Using the \textbf{Christoffel symbol} approach and the spatial inertia formulation:
\[
M_{ij}(\bm{\theta}) = \sum_{l = \max(i,j)}^{n} \bm{\xi}_i^\top\,\mathbf{A}_{li}^\top\,
\hat{\mathbf{M}}_l^s\,\mathbf{A}_{lj}\,\bm{\xi}_j
\]
where $\mathbf{A}_{ij} \in \mathbb{R}^{6\times 6}$ are the adjoint maps between joints:
\[
\mathbf{A}_{ij} = \begin{cases}
\mathrm{Ad}^{-1}_{\mathbf{g}_{ji}}, & i > j \\[3pt]
\mathbf{I}_6, & i = j \\[3pt]
\mathbf{0}, & i < j
\end{cases},\qquad
\mathbf{g}_{ji} = \mathbf{g}_j^{-1}\,\mathbf{g}_i.
\]

\subsection{\texorpdfstring{Computation of $\mathbf{C}(\bm{\theta},\dot{\bm{\theta}})$}{Computation of C(theta, dtheta)}}

The Coriolis matrix is obtained from the partial derivatives of $\mathbf{M}$ via \textbf{Christoffel symbols of the first kind}:
\[
C_{ij} = \sum_{k=1}^{n} \Gamma_{ijk}\,\dot{\theta}_k,\qquad
\Gamma_{ijk} = \frac{1}{2}\left(
\frac{\partial M_{ij}}{\partial\theta_k} +
\frac{\partial M_{ik}}{\partial\theta_j} -
\frac{\partial M_{kj}}{\partial\theta_i}
\right).
\]

The partial derivative $\partial M_{ij}/\partial\theta_k$ is computed analytically:
\begin{align*}
\frac{\partial M_{ij}}{\partial\theta_k} = \sum_{l=\max(i,j)}^{n}
\Bigl[
&\bigl(\mathrm{ad}_{\mathbf{A}_{ki}\bm{\xi}_i}\,\bm{\xi}_k\bigr)^\top\,
\mathbf{A}_{lk}^\top\,\hat{\mathbf{M}}_l^s\,\mathbf{A}_{lj}\,\bm{\xi}_j \\[3pt]
+\;&\bm{\xi}_i^\top\,\mathbf{A}_{li}^\top\,\hat{\mathbf{M}}_l^s\,
\mathbf{A}_{lk}\,\bigl(\mathrm{ad}_{\mathbf{A}_{kj}\bm{\xi}_j}\,\bm{\xi}_k\bigr)
\Bigr],
\end{align*}
valid for $k \le l$, and zero otherwise. Here $\mathrm{ad}_{\bm{\xi}}$ is the $6\times 6$ Lie bracket (adjoint) matrix:
\[
\mathrm{ad}_{\bm{\xi}} = \begin{bmatrix}
\hat{\bm{\omega}} & \hat{\mathbf{v}} \\
\mathbf{0} & \hat{\bm{\omega}}
\end{bmatrix}.
\]

\subsection{\texorpdfstring{Computation of $\mathbf{N}(\bm{\theta},\dot{\bm{\theta}})$}{Computation of N(theta, dtheta)}}

The vector $\mathbf{N}$ consists of two parts:

\subsubsection*{1. Gravity Term}
The gradient of the gravitational potential energy:
\[
\frac{\partial V}{\partial\theta_i} = -\sum_{l=i}^{n} \bm{\xi}_i^\top\,
\mathrm{Ad}_{\mathbf{g}_{0l}}^{-\top}\,\hat{\mathbf{M}}_l^s\,
\mathbf{G}_0
\]
where $\mathbf{G}_0 \in \mathbb{R}^6$ is the \textbf{gravity twist} in the spatial frame:
\[
\mathbf{G}_0 = \begin{bmatrix} -\mathbf{g}_{\text{accel}} \\ \mathbf{0} \end{bmatrix}
= \begin{bmatrix} 0\\0\\9.81\\0\\0\\0 \end{bmatrix}.
\]
(The sign follows the code convention: $-\mathbf{g}_{\text{accel}}$ in the linear part.)

\subsubsection*{2. Viscous Friction Term}
\[
N_{\text{fric},i} = \beta_i\,\dot{\theta}_i
\]
with $\bm{\beta} = [1.0,\;0.5,\;0.4,\;1.0]$ (Nm·s/rad for revolute, N·s/m for prismatic).

\subsubsection*{Total:}
\[
\mathbf{N}(\bm{\theta},\dot{\bm{\theta}}) = -\nabla_\theta V(\bm{\theta}) + \mathrm{diag}(\bm{\beta})\,\dot{\bm{\theta}}.
\]

\subsection{Code Implementation}

The dynamics computation is implemented in \texttt{MCN.m} (and its self-contained variant \texttt{MCNsim.m} for Simulink). The \texttt{MCNsim.m} file is used inside Simulink MATLAB Function blocks because it contains \texttt{adjmg} as a local nested function (no external dependencies).

\begin{lstlisting}[language=Matlab, caption={MCNsim.m --- Dynamics computation (self-contained for Simulink)}]
function [Ms, Cs, Ns] = MCNsim(xi, calMp, lM, beta, m4g, theta, dtheta)
    n = length(theta);
    % Step 1: FK --- compute all g_{0i}
    g = cell(1, n+1); g{1} = eye(4);
    for i = 1:n
        g{i+1} = g{i} * expm(hat_se3(xi(:,i)) * theta(i));
    end
    % Step 2: Adj maps A_{ij}
    A = cell(n,n);
    for i = 1:n, for j = 1:n
        if i > j,      A{i,j} = inv(adjmg(inv(g{j+1}) * g{i+1}));
        elseif i == j, A{i,j} = eye(6);
        else,          A{i,j} = zeros(6);
        end
    end, end
    % Step 3: M matrix
    for i = 1:n, for j = 1:n
        M_ij = 0;
        for l = lM(i,j):n
            M_ij = M_ij + xi(:,i)' * A{l,i}' * calMp(:,:,l) * A{l,j} * xi(:,j);
        end
        Ms(i,j) = M_ij;
    end, end
    % Step 4: C matrix via Christoffel symbols
    for i = 1:n, for j = 1:n
        C_ij = 0;
        for k = 1:n
            C_ij = C_ij + 0.5 * (dM(i,j,k) + dM(i,k,j) - dM(k,j,i)) * dtheta(k);
        end
        Cs(i,j) = C_ij;
    end, end
    % Step 5: N vector (gravity + friction)
    grad_V = compute_gravity_potential_gradient(xi, calMp, g, n);
    Ns = -grad_V + beta(:) .* dtheta(:);
end
\end{lstlisting}

\subsection{\texorpdfstring{The Coupling Matrix $l_M$}{The Coupling Matrix lM}}

The \texttt{lM} matrix encodes which links contribute to each $(i,j)$ pair of the mass matrix:
\[
l_M(i,j) = \max(i,j).
\]
For $n=4$:
\[
l_M = \begin{bmatrix}
1 & 2 & 3 & 4 \\
2 & 2 & 3 & 4 \\
3 & 3 & 3 & 4 \\
4 & 4 & 4 & 4
\end{bmatrix}.
\]

\section{Part (a): Computed Torque Controller}
% ===================================================================

\subsection{Control Law}

The \textbf{computed torque} (feedback linearization) controller applies:
\[
\boxed{\bm{\tau} = \mathbf{M}(\bm{\theta})\,\ddot{\bm{\theta}}^* + \mathbf{C}(\bm{\theta},\dot{\bm{\theta}})\,\dot{\bm{\theta}} + \mathbf{N}(\bm{\theta},\dot{\bm{\theta}})}
\]
where $\ddot{\bm{\theta}}^*$ is a virtual input chosen to achieve tracking. Substituting into the dynamics yields the \textbf{double-integrator} error dynamics:
\[
\ddot{\bm{\theta}} = \ddot{\bm{\theta}}^*.
\]

A \textbf{PD+feedforward} law for $\ddot{\bm{\theta}}^*$ is:
\[
\boxed{\ddot{\bm{\theta}}^* = \ddot{\bm{\theta}}_d + \mathbf{K}_v\bigl(\dot{\bm{\theta}}_d - \dot{\bm{\theta}}\bigr) + \mathbf{K}_p\bigl(\bm{\theta}_d - \bm{\theta}\bigr)}
\]
where $\mathbf{K}_p = \mathrm{diag}(k_{p1},\ldots,k_{p4})$ and $\mathbf{K}_v = \mathrm{diag}(k_{v1},\ldots,k_{v4})$ are positive definite gain matrices.

This yields the \textbf{linear error dynamics}:
\[
\ddot{\mathbf{e}} + \mathbf{K}_v\,\dot{\mathbf{e}} + \mathbf{K}_p\,\mathbf{e} = \mathbf{0},
\qquad \mathbf{e}(t) = \bm{\theta}_d(t) - \bm{\theta}(t).
\]

\subsection{Choosing Gains}

For a desired closed-loop natural frequency $\omega_n$ and damping ratio $\zeta$:
\[
k_{pi} = \omega_{n,i}^2,\qquad k_{vi} = 2\zeta_i\omega_{n,i}.
\]

Suggested initial values:
\[
\mathbf{K}_p = \mathrm{diag}(2500,\;2500,\;2500,\;1600),\qquad
\mathbf{K}_v = \mathrm{diag}(100,\;100,\;100,\;80)
\]
(corresponding to $\omega_n \approx 50$\,rad/s for revolute, 40 for prismatic, with $\zeta=1$).

\subsection{The Rounded Step Trajectory}

Since computed torque control requires $\bm{\theta}_d$ to be \textbf{twice differentiable}, a \textbf{quintic polynomial} rounded step is used:
\[
r(t) = \begin{cases}
0, & t \le t_0 \\[3pt]
10\left(\dfrac{t-t_0}{\Delta t}\right)^3 - 15\left(\dfrac{t-t_0}{\Delta t}\right)^4 + 6\left(\dfrac{t-t_0}{\Delta t}\right)^5, & t_0 < t < t_0+\Delta t \\[3pt]
1, & t \ge t_0 + \Delta t
\end{cases}
\]

\begin{conceptbox}[Properties of the Quintic Rounded Step]
\begin{itemize}
  \item $r(t_0) = 0$, $r(t_0+\Delta t) = 1$
  \item $\dot{r}(t_0) = \dot{r}(t_0+\Delta t) = 0$
  \item $\ddot{r}(t_0) = \ddot{r}(t_0+\Delta t) = 0$
  \item The function is $\mathcal{C}^2$ continuous everywhere.
\end{itemize}
\end{conceptbox}

The transition parameters (from \texttt{initialize\_simulink.m}):
\[
t_0 = 1.0\,\text{s},\qquad \Delta t = 0.2\,\text{s},
\]
so the step starts at $t=1.0$\,s, ramps from $t=1.0$ to $t=1.2$\,s, then holds constant.

The desired joint trajectories are obtained by scaling:
\[
\bm{\theta}_d(t) = \begin{bmatrix} 2 \\ 1 \\ 0.5 \\ 0.08 \end{bmatrix} \cdot r(t),\qquad
\dot{\bm{\theta}}_d(t) = \begin{bmatrix} 2 \\ 1 \\ 0.5 \\ 0.08 \end{bmatrix} \cdot \dot{r}(t),\qquad
\ddot{\bm{\theta}}_d(t) = \begin{bmatrix} 2 \\ 1 \\ 0.5 \\ 0.08 \end{bmatrix} \cdot \ddot{r}(t).
\]

\subsection{Simulink Implementation}

\begin{stepbox}[Step-by-Step Simulink Construction]
\begin{enumerate}[leftmargin=*, label=\textbf{S\arabic*}., ref=S\arabic*]
  \item \textbf{Copy the rounded step block} from \texttt{MLS4\_4\_rounded\_step\_inputR2024a.slx} into your new model. This block outputs $[y,\;\dot{y},\;\ddot{y}]^\top$ where $y$ is the quintic rounded step.
  \item \textbf{Scale the outputs:} Multiply the $y$, $\dot{y}$, $\ddot{y}$ signals by the amplitude vector $[2,\,1,\,0.5,\,0.08]$ to obtain $\bm{\theta}_d$, $\dot{\bm{\theta}}_d$, $\ddot{\bm{\theta}}_d$.

  \item \textbf{Implement two MATLAB Function blocks} for dynamics:
  \begin{itemize}
    \item \textbf{Controller dynamics block:} Contains \texttt{MCNsim} with \textbf{nominal} parameters (call it \texttt{MCN\_ctrl}). Inputs: $\bm{\theta}$, $\dot{\bm{\theta}}$. Outputs: $\mathbf{M}_c$, $\mathbf{C}_c$, $\mathbf{N}_c$.
    \item \textbf{Plant dynamics block:} Contains \texttt{MCNsim} with (potentially perturbed) parameters (call it \texttt{MCN\_plant}). Inputs: $\bm{\theta}$, $\dot{\bm{\theta}}$. Outputs: $\mathbf{M}_p$, $\mathbf{C}_p$, $\mathbf{N}_p$.
  \end{itemize}
  (Initially, both use identical nominal parameters.)

  \item \textbf{Compute $\ddot{\bm{\theta}}^*$:}
  \[
  \ddot{\bm{\theta}}^* = \ddot{\bm{\theta}}_d + \mathbf{K}_v(\dot{\bm{\theta}}_d - \dot{\bm{\theta}}) + \mathbf{K}_p(\bm{\theta}_d - \bm{\theta})
  \]
  Use matrix gain blocks or a MATLAB Function block.

  \item \textbf{Compute control torque $\bm{\tau}$:}
  \[
  \bm{\tau} = \mathbf{M}_c\,\ddot{\bm{\theta}}^* + \mathbf{C}_c\,\dot{\bm{\theta}} + \mathbf{N}_c
  \]

  \item \textbf{Solve forward dynamics} using the plant model:
  \[
  \ddot{\bm{\theta}} = \mathbf{M}_p^{-1}\bigl(\bm{\tau} - \mathbf{C}_p\,\dot{\bm{\theta}} - \mathbf{N}_p\bigr)
  \]
  Use a MATLAB Function block that inverts $\mathbf{M}_p$ and computes the acceleration.

  \item \textbf{Integrate twice} to get $\dot{\bm{\theta}}$ and $\bm{\theta}$:
  \begin{itemize}
    \item First integrator: $\ddot{\bm{\theta}} \to \dot{\bm{\theta}}$
    \item Second integrator: $\dot{\bm{\theta}} \to \bm{\theta}$
  \end{itemize}

  \item \textbf{Initial conditions:} Set $\bm{\theta}(0) = [\pi/4,\;\pi/6,\;\pi/2.5,\;-0.04]^\top$ and $\dot{\bm{\theta}}(0)=\mathbf{0}$.
\end{enumerate}
\end{stepbox}

\section{Part (b): Robustness Study}
% ===================================================================

\subsection{Methodology}

To assess robustness, we introduce a \textbf{mismatch} between the \textbf{plant} (actual physical robot) and the \textbf{controller} (which assumes nominal parameters).

\begin{conceptbox}[Perturbation Strategy]
\textbf{Plant only:} Multiply all entries of each link's spatial inertia matrix $\hat{\mathbf{M}}_i^s$ by $(1+\delta)$ in the MCN\_plant block, where $\delta \in \{-0.10,\,-0.05,\,+0.05,\,+0.10\}$.

\textbf{Controller:} Keep the MCN\_ctrl block using the original (nominal) $\hat{\mathbf{M}}_i^s$.
\end{conceptbox}

\subsection{Implementation in Simulink}

In the \texttt{MCNsim} function used by the plant, modify the \texttt{calMp} array as:
\begin{lstlisting}[language=Matlab]
% In MCN_plant only:
delta = 0.05;  % or -0.05, 0.10, -0.10
calMp_perturbed = calMp * (1 + delta);
% Use calMp_perturbed in the dynamics computation
\end{lstlisting}

Alternatively, pass a \texttt{delta} parameter as an additional input to your MATLAB Function block.

\subsection{What to Observe and Report}

\begin{enumerate}[leftmargin=*]
  \item \textbf{Tracking error:} Plot $\mathbf{e}(t)$ for $\delta=0$ (baseline) vs.\ $\delta=\pm 5\%$, $\pm 10\%$. The error should \textbf{not converge to zero} when there is parameter mismatch --- this is the key weakness of computed torque control (it relies on exact model cancellation).
  \item \textbf{Steady-state error:} Even at steady state, a non-zero error persists due to incorrect gravity compensation.
  \item \textbf{Transient behavior:} Overshoot and settling time change with parameter perturbation.
  \item \textbf{Effect direction:} Positive $\delta$ (heavier robot than modeled) vs.\ negative $\delta$ (lighter robot) produce different error patterns.
\end{enumerate}

\begin{resultbox}[Expected Results]
\begin{itemize}
  \item With \textbf{no perturbation} ($\delta = 0$): perfect asymptotic tracking, $\mathbf{e}(t) \to \mathbf{0}$.
  \item With \textbf{perturbation} ($\delta \neq 0$): non-zero steady-state error; the larger $|\delta|$, the larger the error.
  \item The prismatic joint (joint 4) is typically most sensitive due to direct gravity loading.
\end{itemize}
\end{resultbox}

\section{Part (c): PD Control Scheme}
% ===================================================================

\subsection{PD Control Law}

The \textbf{PD+gravity compensation} controller is:
\[
\boxed{\bm{\tau}_{\text{PD}} = \mathbf{K}_p\bigl(\bm{\theta}_d - \bm{\theta}\bigr) + \mathbf{K}_v\bigl(\dot{\bm{\theta}}_d - \dot{\bm{\theta}}\bigr) + \mathbf{N}(\bm{\theta})}
\]

Compared to computed torque, the PD controller:
\begin{itemize}
  \item \textbf{Does not use} $\mathbf{M}(\bm{\theta})$ or $\mathbf{C}(\bm{\theta},\dot{\bm{\theta}})$ in the feedback path (only $\mathbf{N}$ for gravity compensation).
  \item Is \textbf{simpler} to implement and computationally cheaper.
  \item Has \textbf{nonlinear} closed-loop error dynamics (cannot perfectly cancel nonlinearities).
  \item \textbf{Does not require} $\ddot{\bm{\theta}}_d$ (only $\bm{\theta}_d$ and $\dot{\bm{\theta}}_d$).
\end{itemize}

The closed-loop dynamics under PD control:
\[
\mathbf{M}(\bm{\theta})\,\ddot{\bm{\theta}} + \mathbf{C}(\bm{\theta},\dot{\bm{\theta}})\,\dot{\bm{\theta}} + \mathbf{N}(\bm{\theta}) = \mathbf{K}_p\mathbf{e} + \mathbf{K}_v\dot{\mathbf{e}} + \mathbf{N}(\bm{\theta})
\]
which does \textbf{not} reduce to a simple linear system.

\subsection{Simulink Implementation}

\begin{stepbox}[PD Controller Block Diagram --- Simpler than computed torque]
\begin{enumerate}
  \item Remove the $\mathbf{M}_c$, $\mathbf{C}_c$, and $\mathbf{N}_c$ blocks.
  \item Remove the feedforward $\ddot{\bm{\theta}}_d$ term.
  \item Compute:
  \[
  \bm{\tau} = \mathbf{K}_p(\bm{\theta}_d - \bm{\theta}) + \mathbf{K}_v(\dot{\bm{\theta}}_d - \dot{\bm{\theta}}) + \mathbf{N}_p(\bm{\theta})
  \]
  where $\mathbf{N}_p$ is the gravity vector only (friction optional). Note: for PD only (no gravity comp), omit the $\mathbf{N}_p$ term.
  \item Feed $\bm{\tau}$ directly to the forward dynamics (plant model).
\end{enumerate}
\end{stepbox}

\subsection{Gain Tuning for PD}

PD gains typically need to be \textbf{larger} than computed torque gains to compensate for the lack of model-based decoupling:
\[
\mathbf{K}_p^{\text{PD}} \approx (2\text{--}5) \times \mathbf{K}_p^{\text{CT}},\qquad
\mathbf{K}_v^{\text{PD}} \approx (2\text{--}5) \times \mathbf{K}_v^{\text{CT}}.
\]

\subsection{What to Compare and Report}

\begin{table}[H]
\centering
\caption{Comparison: Computed Torque vs.\ PD Control}
\begin{tabular}{p{3.5cm}p{6cm}p{5cm}}
\toprule
\textbf{Criterion} & \textbf{Computed Torque} & \textbf{PD (+ Gravity Comp)} \\
\midrule
Model dependency & Full $\mathbf{M},\mathbf{C},\mathbf{N}$ & $\mathbf{N}$ only (or none) \\
Computational cost & High (matrix inversion) & Low \\
Tracking accuracy & Excellent (if model exact) & Moderate (nonlinear coupling) \\

\bottomrule
\end{tabular}
\end{table}

\begin{resultbox}[Expected Results]
\begin{itemize}
  \item \textbf{Computed torque} achieves faster, more accurate tracking when the model is exact.
  \item \textbf{PD} has larger tracking errors during transients due to uncompensated dynamic coupling.
  \item \textbf{PD} is more robust to parameter uncertainty (the error degrades gracefully rather than catastrophically).
  \item With gravity compensation, PD achieves zero steady-state error for step references (by the passivity argument).
  \item Without gravity compensation, PD has steady-state error.
\end{itemize}
\end{resultbox}

\section{Part (d): Torque Saturation}
% ===================================================================

\subsection{Problem Formulation}

Real actuators have finite torque/force capabilities. The saturation limits are:
\[
|\tau_i| \le \tau_{i,\max},\qquad
\bm{\tau}_{\max} = \begin{bmatrix} 6000 \\ 2000 \\ 1000 \\ 100 \end{bmatrix}
\;\text{Nm (joints 1--3) / N (joint 4)}.
\]

When the computed torque controller demands $\tau_i$ beyond $\tau_{i,\max}$, the actual applied torque is \textbf{clipped}:
\[
\tau_i^{\text{applied}} = \mathrm{sat}(\tau_i^{\text{commanded}};\;-\tau_{i,\max},\;\tau_{i,\max})
= \begin{cases}
\tau_{i,\max}, & \tau_i^{\text{commanded}} > \tau_{i,\max} \\
\tau_i^{\text{commanded}}, & |\tau_i^{\text{commanded}}| \le \tau_{i,\max} \\
-\tau_{i,\max}, & \tau_i^{\text{commanded}} < -\tau_{i,\max}
\end{cases}
\]

\subsection{Simulink Implementation}

\begin{stepbox}[Saturation Block Placement]
Insert a \textbf{Saturation} block (from Simulink $\to$ Discontinuities library) between the computed torque output and the forward dynamics plant input, with limits set to $\pm[6000,\,2000,\,1000,\,100]$.
\end{stepbox}

\subsection{Effects of Saturation}

\begin{conceptbox}[Key Phenomena]
\begin{enumerate}
  \item \textbf{Integrator windup:} When the torque saturates, the error keeps accumulating in the integrator (if used), causing large overshoots when the system comes out of saturation. This is called \textbf{windup}.
  \item \textbf{Slower response:} The system cannot accelerate as fast as commanded, so tracking is degraded during the transient.
  \item \textbf{Potential instability:} If the controller relies on the exact cancellation of nonlinearities and the applied torque differs significantly from the commanded torque, the error dynamics are no longer linear and may become unstable.
  \item \textbf{Step-size dependence:} Larger desired steps require larger torques; the saturation limits may be exceeded more severely.
\end{enumerate}
\end{conceptbox}

\subsection{Anti-Windup Strategies (Optional Investigation)}

\begin{enumerate}[leftmargin=*]
  \item \textbf{Back-calculation:} Feed the difference $(\bm{\tau}_{\text{applied}} - \bm{\tau}_{\text{commanded}})$ back to reduce the integral action.
  \item \textbf{Conditional integration:} Freeze the integrator when saturation is active.
  \item \textbf{Reference governor:} Scale down $\ddot{\bm{\theta}}^*$ when the commanded torque exceeds limits.
\end{enumerate}

\subsection{What to Report}

\begin{enumerate}[leftmargin=*]
  \item Plot commanded vs.\ applied torques for each joint.
  \item Compare tracking performance with and without saturation.
  \item Identify which joints saturate first (typically joint 1, which supports the entire arm).
  \item Investigate whether the saturation limits are adequate for the desired trajectory.
  \item If windup is observed, suggest and test an anti-windup modification.
\end{enumerate}

\begin{resultbox}[Expected Results]
\begin{itemize}
  \item Joint 1 (the base joint) requires the largest torque and is most likely to saturate first.
  \item Under saturation, tracking errors increase and may not converge to zero.
  \item The response becomes slower and may exhibit overshoot due to windup.
  \item With anti-windup, the overshoot is reduced but the speed is still limited by actuator capability.
\end{itemize}
\end{resultbox}

\end{conceptbox}

\section{Troubleshooting Tips}
% ===================================================================

\begin{warningbox}[Common Pitfalls]
\begin{enumerate}[leftmargin=*]
  \item \textbf{Algebraic loop:} If $\mathbf{M}$, $\mathbf{C}$, or $\mathbf{N}$ depend on $\ddot{\bm{\theta}}$, Simulink will report an algebraic loop. Ensure your MCN function takes only $(\bm{\theta},\dot{\bm{\theta}})$ as inputs (not $\ddot{\bm{\theta}}$).
  \item \textbf{Dimension mismatch:} All signals are $4\times 1$ vectors. Use \texttt{Selector} blocks or \texttt{Demux}/\texttt{Mux} blocks carefully.
  \item \textbf{Singular $\mathbf{M}$:} $\mathbf{M}(\bm{\theta})$ is always invertible for this manipulator. If you get a singular matrix warning, check your inertia parameters.
  \item \textbf{MATLAB Function block dependencies:} Use \texttt{MCNsim.m} (self-contained) inside Simulink MATLAB Function blocks, not \texttt{MCN.m} (has external dependencies).
  \item \textbf{Step size too large:} If the solver diverges, reduce the maximum step size in Configuration Parameters (try 1e-3 or smaller).
  \item \textbf{Initial conditions:} The initial $\bm{\theta}$ must be set in the integrator blocks. The initial $\dot{\bm{\theta}}$ is typically zero.
\end{enumerate}
\end{warningbox}

\section{Conclusion}
% ===================================================================

This homework demonstrates the full pipeline of robot control system design:

\begin{enumerate}[leftmargin=*]
  \item \textbf{Kinematic and dynamic modeling} using screw theory (product of exponentials) provides an elegant, coordinate-free formulation suitable for symbolic and numerical computation.
  \item \textbf{Computed torque control} achieves perfect tracking when the model is exact, by exactly canceling the nonlinear dynamics and imposing linear error dynamics.
  \item \textbf{Robustness} is the Achilles' heel of computed torque control --- parameter mismatch leads to non-zero tracking errors, motivating adaptive or robust control extensions.
  \item \textbf{PD control} is simpler and more robust but sacrifices tracking performance due to uncompensated dynamic coupling.
  \item \textbf{Actuator saturation} imposes practical limits that must be accounted for, potentially requiring anti-windup strategies.
\end{enumerate}
